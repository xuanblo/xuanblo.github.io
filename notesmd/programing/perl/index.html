    hash哈希¶

    my %hash = (
    "key1" => ['a', 'b'], # 哈希的数组
    "key2" => "item",
    "key3" => (           # 二维哈希
        'kkey1' => [1, 2 ,3],
        'kkey2' => [1, 2]
        )
    );

# 遍历哈希, 对key, value排序
foreach my $key (sort{$a cmp $b} keys %hash)
{
    # 按key值字符排序
}
foreach my $key (sort{$a <=> $b} keys %hash)
{
    # 按key值数字排序
}
foreach my $key (sort{$hash{$a} <=> $hash{$b}} keys %hash)
{
    # 按value值数字排序
}
